{# Renders a (potentially nested) mission list. `items` is an array whose entries are either: - a string (markdown inline) - an object {text, items} where `items` is another such array. Nesting is supported up to 3 levels — deeper levels are flattened. #} {% macro mission_list(items) %}
    {% for entry in items %} {% if entry is string %}
  • {{ entry | markdown(inline=true) | safe }}
    • {% else %}
  • {{ entry.text | markdown(inline=true) | safe }} {% if entry.items %}
      {% for sub in entry.items %} {% if sub is string %}
    • {{ sub | markdown(inline=true) | safe }}
      • {% else %}
    • {{ sub.text | markdown(inline=true) | safe }} {% if sub.items %}
        {% for leaf in sub.items %} {% if leaf is string %}
      • {{ leaf | markdown(inline=true) | safe }}
        • {% else %}
      • {{ leaf.text | markdown(inline=true) | safe }}
        • {% endif %} {% endfor %}
      {% endif %}
      • {% endif %} {% endfor %}
    {% endif %}
    • {% endif %} {% endfor %}
{% endmacro %}
